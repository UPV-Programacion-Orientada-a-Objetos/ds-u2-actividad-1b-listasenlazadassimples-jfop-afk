26float cadenaAFloat(const char* cadena) {
27 float resultado = 0.0;
28 float decimal = 0.0;
29 int i = 0;
30 bool negativo = false;
31 bool enDecimal = false;
32 float divisor = 10.0;
33
34 if (cadena[0] == '-') {
35 negativo = true;
36 i = 1;
37 }
38
39 while (cadena[i] != '\0') {
40 if (cadena[i] == '.') {
41 enDecimal = true;
42 i = i + 1;
43 continue;
44 }
45
46 int digito = cadena[i] - '0';
47
48 if (!enDecimal) {
49 resultado = resultado * 10.0 + digito;
50 } else {
51 decimal = decimal + digito / divisor;
52 divisor = divisor * 10.0;
53 }
54
55 i = i + 1;
56 }
57
58 resultado = resultado + decimal;
59
60 if (negativo) {
61 resultado = -resultado;
62 }
63
64 return resultado;
